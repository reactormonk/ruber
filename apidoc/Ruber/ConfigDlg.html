def create_data
  pages = Hash.new{|h, k| h[k] = []}
  groups = Dictionary.alpha
  Ruber[:components].each_component do |c|
    c.plugin_description.config_widgets.each do |i|
      groups[i.caption] ||= i.pixmap
      w = i.code ? eval(i.code, TOPLEVEL_BINDING) : i.class_obj.new
      @widgets << w
      pages[i.caption] << w
    end
  end
  groups.each_pair do |k, v|
    page = Qt::Widget.new
    page.layout = Qt::VBoxLayout.new page
    pages[k].each do |w|
      w.parent = page
      page.layout.add_widget w
    end
    add_page page, k, v
  end
  
end

- (Object) sizeHint

Informs the dialog of a page which it will need to add when its update_pages method will be called. =Arguments widget:: the widget representing the page. Note that it won’t be directly inserted in the dialog: a container widget will be created and widget will be made its children. The new widget will be inserted in the dialog name_:: the label to display in the navigation section of the dialog. It must be unique unless allowduplicate is true pixmap_:: the path of the pixmap to show in the navigation part of the dialog allowduplicate:: what to do if a page with name name is given. If false or nil, an ArgumentError exception will be raised. Otherwise, the widget will be added at the bottom of the existing page.
